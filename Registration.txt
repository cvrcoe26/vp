Open Visual Studio → Create ASP.NET Web Application (.NET Framework) → Choose Web Forms → Click Create.

Add a new Web Form → Name it Registration.aspx.

<%@ Page Language="C#" AutoEventWireup="true" CodeBehind="Registration.aspx.cs" Inherits="CollegeInfoApp.Registration" %>

<!DOCTYPE html>
<html>
<head>
    <title>Registration Page</title>
</head>
<body>
    <form id="form1" runat="server">
        <h2>Registration Form</h2>

        Name: <asp:TextBox ID="txtName" runat="server"></asp:TextBox><br /><br />
        Email: <asp:TextBox ID="txtEmail" runat="server"></asp:TextBox><br /><br />
        Password: <asp:TextBox ID="txtPassword" TextMode="Password" runat="server"></asp:TextBox><br /><br />

        <asp:Button ID="btnSubmit" Text="Submit" runat="server" OnClick="btnSubmit_Click" /><br /><br />

        <asp:Label ID="lblDisplay" runat="server"></asp:Label>
    </form>
</body>
</html>


using System;

namespace CollegeInfoApp
{
    public partial class Registration : System.Web.UI.Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {
        }

        protected void btnSubmit_Click(object sender, EventArgs e)
        {
            string name = txtName.Text;
            string email = txtEmail.Text;
            string password = txtPassword.Text;

            // Display entered data
            lblDisplay.Text = "You entered:<br/>" +
                              "Name: " + name + "<br/>" +
                              "Email: " + email + "<br/>" +
                              "Password: " + password;
        }
    }
}
