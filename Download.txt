Open Visual Studio → Create a new project → Choose
“Windows Forms App (.NET Framework)” → Click Create.

Design the Form

Add:

1 TextBox → give id: txtUrl → (for entering the file URL)

1 Button → Text: “Download”

1 Label → Text: “Enter file URL:”

Double-click the Download button to open the Click event, and paste the below code inside the btnDownload_Click function:

        private void btnDownload_Click(object sender, EventArgs e)
        {
            try
            {
                string url = textUrl.Text;
                string savePath = @"C:\Users\CVR\Downloads\vp" + System.IO.Path.GetFileName(url);
                using (WebClient client = new WebClient())
                 {
                     client.DownloadFile(url, savePath);
                     Label1.Text = @"file uploaded to " + savePath;
                 }
            }
            catch (Exception ex)
            {
               Label1.Text = "Error downloading file";
            }
        }
