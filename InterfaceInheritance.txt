using System;
interface IFirst {
  void Show();
}
interface ISecond {
  void Show();
}
class Demo: IFirst,
ISecond {
  void IFirst.Show() {
    Console.WriteLine("IFirst Show");
  }
  void ISecond.Show() {
    Console.WriteLine("ISecond Show");
  }
}
class Program {
  static void Main() {
    Demo d = new Demo();
    ((IFirst) d).Show();
    ((ISecond) d).Show();
  }
}
