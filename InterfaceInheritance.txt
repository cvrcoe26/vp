using System;

interface IFirst
{
    void ShowFirst();
}

interface ISecond : IFirst
{
    void ShowSecond();
}

class Demo : ISecond
{
    void IFirst.ShowFirst()
    {
        Console.WriteLine("IFirst Show");
    }

    void ISecond.ShowSecond()
    {
        Console.WriteLine("ISecond Show");
    }
}

class Program
{
    static void Main()
    {
        Demo d = new Demo();

        IFirst f = d;
        f.ShowFirst();

        ISecond s = d;
        s.ShowSecond();
    }
}
