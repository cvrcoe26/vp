
Open Visual Studio → File → New → Website → ASP.NET Web Site
Add a new page → EmailPhoneValidation.aspx
Design the Form (EmailPhoneValidation.aspx)
<%@ Page Language="C#" AutoEventWireup="true" CodeFile="EmailPhoneValidation.aspx.cs" Inherits="EmailPhoneValidation" %>

<!DOCTYPE html>
<html>
<head runat="server">
    <title>Email and Phone Validation</title>
</head>
<body>
    <form id="form1" runat="server">
        <h2>Email and Phone Validation</h2>
        <table>
            <tr>
                <td>Email:</td>
                <td>
                    <asp:TextBox ID="txtEmail" runat="server" />
                    <asp:RegularExpressionValidator 
                        ID="revEmail" runat="server"
                        ControlToValidate="txtEmail"
                        ErrorMessage="Invalid Email Format"
                        ForeColor="Red"
                        ValidationExpression="^[^@\s]+@[^@\s]+\.[^@\s]+$" />
                </td>
            </tr>
            <tr>
                <td>Phone:</td>
                <td>
                    <asp:TextBox ID="txtPhone" runat="server" />
                    <asp:RegularExpressionValidator 
                        ID="revPhone" runat="server"
                        ControlToValidate="txtPhone"
                        ErrorMessage="Invalid Phone Number"
                        ForeColor="Red"
                        ValidationExpression="^[0-9]{10}$" />
                </td>
            </tr>
            <tr>
                <td colspan="2">
                    <asp:Button ID="btnSubmit" runat="server" Text="Submit" />
                </td>
            </tr>
        </table>

        <asp:ValidationSummary ID="ValidationSummary1" runat="server" ForeColor="Red" />
    </form>
</body>
</html>

Web.config Setting

<configuration>
<appSettings>
<add key="ValidationSettings:UnobtrusiveValidationMode" value="None" />
</appSettings>
</configuration>

Run the Application
Press F5 in Visual Studio.
